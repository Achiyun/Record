登录:http://localhost:8080/day36/userServlet?action=login

激活:http://localhost:8080/day36/userServle?action=active

每一个模块对应一个Servlet,发现doGet()方法里面,代码都是重复的,所以抽取一个通用的BaseServlet基类, 让各个模块Servlet继承BaseServlet.通用的BaseServlet

public class BaseServlet extends HttpServlet {
    @Override
    protected void service(HttpServletRequest req, HttpServletResponse resp) {
        // 获取请求路径
        String uri = req.getRequestURI();
        System.out.println("BaseServlet uri = " + uri);
        // 获取uri最后的一节方法名字符串
        String methodName = uri.substring(uri.lastIndexOf("/") + 1);
        System.out.println("methodName = " + methodName);
        // 使用反射的方式调用方法
        //Class aClass = this.getClass();
        //System.out.println("aClass = " + aClass);
        try {
            Method method = this.getClass().getMethod(methodName, 
                                         HttpServletRequest.class,
                                         HttpServletResponse.class);
            method.invoke(this,req,resp);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

# 小结

  1. 把公共的代码抽取到父类BaseServlet, 模块的Servlet继承BaseServlet就可以了

  2. BaseServlet里面的异常打印不要删除

  3. 在BaseServlet里面反射的API是getMethod 也就意味着模块里面的Servlet的方法应该是public的

# 二、自定义MVC框架

# 1. BaseServlet问题分析

​ BaseServlet中，可以灵活的处理客户端的请求，应对部分项目开发没有问题。

​ 这种实现要求客户端处理请求的Servlet也必须要继承父类BaseServlet，因为是在父类的service()中完成请求解析与调用。这种方式不是很便捷并且耦合度比较高。 ​ 接下来想进一步重构BaseServlet，让其用起来更方便更便捷，让它们的耦合关系更松散，比如创建一个总控制器类，这个总控制器继承HttpServlet类，其他的控制器（子控制器）是普通Java类，不需要间接继承HttpServlet，然后由总控制器动态的去调用子控制器的业务方法，这种动态调用是采用在参数中不加入action的方式来实现的。

如下:

public class UserController {

    @RequestMapping("/user/add")
    public void add(HttpServletRequest request, HttpServletResponse response) throws IOException {
        response.getWriter().write("add.....");
    }
}

# 2. 自定义SpringMVC框架初级版本

# 2.1 思路

  1. 创建@RequestMapping注解

  2. 创建UserController, 定义方法, 在这个方法上面添加@RequestMapping注解

  3. 创建DispatcherServlet继承HttpServlet, 路径配置*.do

  4. 在DispatcherServlet的service()方法里面

    1.获得请求的URI和项目部署路径, 截取获得映路径 eg: /user/login 2.扫描某个包里面的所有类的字节码对象集合List 3.遍历字节码对象集合List 4.获得类里面的所有的Method 5.遍历所有的Method 6.获得method上面的RequestMapping注解 获得注解的value属性值 7.判断value属性值是否和获得映路径一致, 一致 就调用method

# 2.2 代码实现

  • RequestMapping注解

public @interface RequestMapping {

@WebServlet("*.do")
public class DispatcherServlet extends HttpServlet {
    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        doGet(request, response);
    }

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        //1. 获取请求的uri: /项目名/user/xxx.do
        String uri = request.getRequestURI();
        //2. 去掉"/项目名"和".do"
        //2.1 获取"/项目名"
        String contextPath = request.getContextPath();
        //2.2 获取到用于匹配的路径
        String mappingPath = uri.substring(contextPath.length(), uri.lastIndexOf("."));

        //3.扫描Controller所在的整个包中的所有类: 指定包下的所有类的字节码对象
        List<Class<?>> classList = ClassScannerUtils.getClasssFromPackage("com.tilaiedu.controller");
        //4. 遍历出每一个字节码对象
        try {
            if (classList != null) {
                for (Class<?> clazz : classList) {
                    //获取该字节码对象对应的类中的所有public方法
                    Method[] methods = clazz.getMethods();
                    //遍历出每一个公有方法
                    for (Method method : methods) {
                        //判断该方法上是否有RequestMapping注解
                        if (method.isAnnotationPresent(RequestMapping.class)) {
                            //方法上有RequestMapping注解，则获取该注解的值
                            String requestMappingValue = method.getAnnotation(RequestMapping.class).value();

                            //用mappingPath匹配RequestMappingValue
                            if (mappingPath.equals(requestMappingValue)) {
                                //匹配上了，则调用该方法
                                method.invoke(clazz.newInstance(),request,response);
                                return;
                            }
                        }
                    }
                }
            }

            //说明请求，没有对应的方法去处理
            throw new RuntimeException("No Method To Invoke.....");
        } catch (Exception e) {
            e.printStackTrace();
            throw new RuntimeException(e.getMessage());
        }
    }
}

# 3. 自定义SpringMVC框架终极版本

初级版本存在问题:

  1. 扫描包的代码不应该放在service里面，因为放在service里面的话，会每次处理请求都执行
  2. DispatcherServlet的映射路径是使用注解配置的
  3. 要扫描的Controller的包名写死了
  4. 每次执行Controller中的方法都会新创建Controller对象
  5. 会扫描整个controller包中的所有类，我们只想扫描能够处理请求的那些类

# 3.1 解决问题的思路思路

  1. 将扫描包的代码写在DispatcherServlet的初始化方法init()里面
  2. 不在DispatcherServlet中使用注解配置映射路径，而由使用者自己在web.xml中配置映射路径
  3. 在web.xml配置文件中，使用servlet的初始化参数进行配置
  4. 使用单例(容器式单例)
  5. 定义一个Controller注解，添加到能够处理请求的类上，扫描的时候我就只需要判断类上是否有Controller注解就行了

import java.io.Serializable;
import java.lang.reflect.Method;

public class MvcMethod implements Serializable{
    private Object object;
    private Method method;

    public MvcMethod() {
    }

    public MvcMethod(Object object, Method method) {
        this.object = object;
        this.method = method;
    }

    @Override
    public String toString() {
        return "MvcMethod{" +
                "object=" + object +
                ", method=" + method +
                '}';
    }

    public Object getObject() {
        return object;
    }

    public void setObject(Object object) {
        this.object = object;
    }

    public Method getMethod() {
        return method;
    }

    public void setMethod(Method method) {
        this.method = method;
    }
}

/**
 * 一个负责出来请求和相应的总控制器类，可以处理当前项目的所有请求和响应
 * 让后分发给自定义的子控制器类
 * object对象(Controller类的对象)和method和requestMappingValue是一一对应关系，所以将他们仨属性封装到一个对象中
 * 一个注解值对应一个MvcMethod, 不同的注解值对应不同的mvcMethod对象，所以可以用Map，注解值作为key，MvcMethod作为value
 */
public class DispatcherServlet extends HttpServlet {

    // 创建一个保存关联控制器类、方法、注解的集合
    private final Map<String,MvcMethod> methodMap = new HashMap<>();

    @Override
    public void init(ServletConfig config) throws ServletException {
        super.init(config);
        // 获取用户配置的要扫描的包
        String scanPackage = config.getInitParameter("scanPackage");
        // 扫描整个controller包
        List<Class<?>> classList = ClassScannerUtils.getClasssFromPackage(scanPackage);

        try {
            // 遍历所有字节码文件，查询方法对应的注解值
            for (Class<?> aClass : classList) {
                // 判断当前类是否有controller注解
                if (!aClass.isAnnotationPresent(Controller.class)) continue;
                // 获取该字节码文件中所有的公共方法
                Method[] methods = aClass.getMethods();
                // 遍历每一个方法
                for (Method method : methods) {
                    // 判断方法上是否有RequestMapping注解
                    if (method.isAnnotationPresent(RequestMapping.class)){
                        // 获取对应的注解值
                        String requestMappingValue = method.getAnnotation(RequestMapping.class).value();
                        // 创建当前类对象
                        Object obj = aClass.getConstructor().newInstance();
                        // 将关联信息和对象保存
                        methodMap.put(requestMappingValue,new MvcMethod(method,obj));
                    }
                }
            }
        } catch (Exception e) {
            throw new RuntimeException(e.getMessage());
        }

    }

    @Override
    protected void service(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        // 获取请求路径
        String uri = req.getRequestURI();
        System.out.println("uri = " + uri);
        // 切割字符串，获取中间的资源路径 /customMVC/aaa.action
        // 获取资源路径
        String contextPath = req.getContextPath();
        // 获取资源名称
        String resourceName = uri.substring(contextPath.length(), uri.lastIndexOf("."));
        System.out.println("resourceName = " + resourceName);
        // 根据资源路径获取对应的类和方法
        MvcMethod mvcMethod = methodMap.get(resourceName);
        try {
            if (mvcMethod != null){
                mvcMethod.getMethod().invoke(mvcMethod.getObj(),req,resp);
                return;
            }
            throw new RuntimeException("---找不到资源路径"+resourceName+"匹配的方法");
        } catch (Exception e) {
            throw new RuntimeException(e.getMessage());
        }
    }
}

  1. 在web.xml中配置DispatcherServlet
    1. 指定映射路径为*.do
    2. 配置servlet标签中的init-param，指定参数名为scanPackage，参数值为要扫描的包名
    3. 通过load-on-startup标签配置DispatcherServlet在服务器启动的时候加载
<servlet>
    <servlet-name>dispatcherServlet</servlet-name>
    <servlet-class>com.tilaiedu.frame.DispatcherServlet</servlet-class>
    <init-param>
        <param-name>scanPackage</param-name>
        <!--要写自己的包名-->
        <param-value>com.xxxx.controller</param-value>
    </init-param>

    <!--配置DispatcherServlet在服务器启动的时候创建-->
    <load-on-startup>1</load-on-startup>
</servlet>
<servlet-mapping>
    <servlet-name>dispatcherServlet</servlet-name>
    <url-pattern>*.do</url-pattern>
</servlet-mapping>
